Concept: has_size¶

Source

  • example-4-concept-has-size.cpp

template <typename V>
requires has_size<V>
double hypotenuse(const V& v)
{
    double sumsq = 0;
    for (size_t i=0; i<v.size(); ++i)
        sumsq += v[i]*v[i];
    return std::sqrt(sumsq);
}
