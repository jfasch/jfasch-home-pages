Duck Typed, Unrelated, But Similar¶

  • Two unrelated classes: Duck, Goose

    • Both quack() and walk() differently, but similar

    • Both live() similarly, too

duck-and-goose.py¶
#!/usr/bin/env python3

class Duck:
    def __init__(self, height):
        self.height = height

    def quack(self):
        print('quackquackquackquackquackquack')
    def walk(self):
        print('watschelwatschelwatschelwatschelwatschelwatschel')

    def live(self):
        self.spend_life(self.height)

    def spend_life(self, num):
        for _ in range(num):
            self.quack()
            self.walk()

class Goose:
    def __init__(self, width):
        self.width = width

    def quack(self):
        print('quackquackquackquackquackquack quackquackquackquackquackquack')
    def walk(self):
        print('watschelwatschelwatschelwatschelwatschelwatschel watschelwatschelwatschelwatschelwatschelwatschel')

    def live(self):
        self.spend_life(self.width)

    def spend_life(self, num):
        for _ in range(num):
            self.quack()
            self.walk()

goose = Goose(10)
duck = Duck(5)

goose.live()
duck.live()

Base Class Attribute: Initialize Base Using super()¶

duck-and-goose-baseattr.py¶
#!/usr/bin/env python3

class GooseOrDuck:
    def __init__(self, num):
        self.num = num

    def live(self):
        for _ in range(self.num):
            self.quack()
            self.walk()


class Duck(GooseOrDuck):
    def __init__(self, height):
        super().__init__(height)

    def quack(self):
        print('quackquackquackquackquackquack')
    def walk(self):
        print('watschelwatschelwatschelwatschelwatschelwatschel')

class Goose(GooseOrDuck):
    def __init__(self, width):
        super().__init__(width)

    def quack(self):
        print('quackquackquackquackquackquack quackquackquackquackquackquack')
    def walk(self):
        print('watschelwatschelwatschelwatschelwatschelwatschel watschelwatschelwatschelwatschelwatschelwatschel')

goose = Goose(10)
duck = Duck(5)

goose.live()
duck.live()
