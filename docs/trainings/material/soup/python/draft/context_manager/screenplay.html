for line in f:
    if line.find('jfasch') >= 0: # exception!
        print(line)
        break
else:
    print('no')

# won't get here
f.close()
print('file closed')

Exception Safety¶

code/10-with-file.py¶
with open('/etc/passwd') as f:
    for line in f:
        if line.find('jfasch') >= 0:
            print(line)
            break
    else:
        print('no')

$ python code/10-with-file.py
