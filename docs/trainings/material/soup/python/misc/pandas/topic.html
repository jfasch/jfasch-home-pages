jupyter-execute¶

import pandas as pd

users = {
    'firstname': ['Joerg', 'Johanna', 'Caro', 'Philipp'],
    'lastname': ['Faschingbauer', 'Faschingbauer', 'Faschingbauer', 'Lichtenberger'],
    'svnr': ['1037190666', '1234110695', '2345250497', '345606041986'],
}

df = pd.DataFrame(users)
