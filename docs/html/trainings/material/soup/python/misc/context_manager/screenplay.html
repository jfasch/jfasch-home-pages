code/05-manual-file.py¶
f = open('/etc/passwd', 'br')   # binary!
for line in f:
    if line.find('jfasch') >= 0: # exception!
        print(line)
        break
else:
    print('no')

# won't get here
f.close()
print('file closed')

Exception Safety¶

code/10-with-file.py¶
with open('/etc/passwd') as f:
    for line in f:
        if line.find('jfasch') >= 0:
            print(line)
            break
    else:
        print('no')

$ python code/10-with-file.py
jfasch:x:1000:1000:Joerg Faschingbauer:/home/jfasch:/bin/bash

orig_print = print
def prefix_print(*args, **kwargs):
    pfargs = ('prefix',) + args
    orig_print(*pfargs, **kwargs)
print = prefix_print
